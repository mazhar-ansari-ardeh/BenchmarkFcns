g(y, J) = 4 \sum_{j \in J} y_j^2 - 2 \prod_{j \in J} \cos\left(\frac{20y_j\pi}{\sqrt{j}}\right) + 2 \\
y_j = x_j - \sin(6\pi x_1 + \frac{j\pi}{n}) \\
J_1 = \{j | j \text{ is odd and } 2 \le j \le n\}, J_2 = \{j | j \text{ is even and } 2 \le j \le n\}
