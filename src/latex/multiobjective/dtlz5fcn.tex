g(\mathbf{x}_M) = \sum_{x_i \in \mathbf{x}_M} (x_i - 0.5)^2 \\
\theta_1 = x_1 \pi/2 \\
\theta_i = \frac{\pi}{4(1+g(\mathbf{x}_M))} (1 + 2g(\mathbf{x}_M) x_i), \quad i=2, \dots, M-1 \\
f_1(\mathbf{x}) = (1 + g(\mathbf{x}_M)) \cos(\theta_1) \dots \cos(\theta_{M-1}) \\
\vdots \\
f_M(\mathbf{x}) = (1 + g(\mathbf{x}_M)) \sin(\theta_1)
