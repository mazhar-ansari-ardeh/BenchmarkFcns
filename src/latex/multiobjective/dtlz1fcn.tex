f_i(\textbf{x}) = 0.5 (1 + g(\textbf{x}_M)) \left( \prod_{j=1}^{M-i} x_j \right) (1 - x_{M-i+1}) \quad \text{for } i = 1, \dots, M \\
g(\textbf{x}_M) = 100 \left( k + \sum_{x_j \in \textbf{x}_M} \left( (x_j - 0.5)^2 - \cos(20 \pi (x_j - 0.5)) \right) \right) \\
\text{where } \textbf{x}_M = [x_{n-k+1}, \dots, x_n] \text{ and } k = n - M + 1 \text{, and } (1 - x_{M-i+1}) \text{ is omitted for } i=1.
