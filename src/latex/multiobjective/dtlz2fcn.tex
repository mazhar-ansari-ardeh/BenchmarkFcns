f_i(\textbf{x}) = (1 + g(\textbf{x}_M)) \left( \prod_{j=1}^{M-i} \cos(x_j \pi / 2) \right) \sin(x_{M-i+1} \pi / 2) \quad \text{for } i = 1, \dots, M \\
g(\textbf{x}_M) = \sum_{x_j \in \textbf{x}_M} (x_j - 0.5)^2 \\
\text{where } \textbf{x}_M = [x_{n-k+1}, \dots, x_n] \text{ and } k = n - M + 1 \text{, and } \sin(x_{M-i+1} \pi / 2) \text{ is omitted for } i=1.
